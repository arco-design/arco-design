## en-US

The simplest usage.

```js
import { Message, Button } from '@arco-design/web-react';

const App = () => {
  return (
    <Button
      onClick={() => {
        Message.info('This is an info message!');
      }}
      type="primary"
    >
      Open Message
    </Button>
  );
};

export default App;
```
