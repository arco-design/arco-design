## zh-CN

用于单元格合并，表头只能进行列合并。

## en-US

Used for cell merging, the header can only be column merged.

```js
import { Table } from '@arco-design/web-react';
const columns = [
  {
    title: 'Name',
    dataIndex: 'name',
    render: (col, item, index) => {
      const obj = {
        children: col,
        props: {},
      };

      if (index > 3) {
        obj.props.colSpan = 2;
      }

      return obj;
    },
  },
  {
    title: 'Salary',
    dataIndex: 'salary',
    render: (col, item, index) => {
      const obj = {
        children: col,
        props: {},
      };

      if (index > 3) {
        obj.props.colSpan = 0;
      }

      return obj;
    },
  },
  {
    title: 'Detail',
    dataIndex: 'address',
    colSpan: 2,
    render: (col, item, index) => {
      const obj = {
        children: col,
        props: {},
      };

      if (index === 0) {
        obj.props.rowSpan = 3;
      }

      if (index === 1 || index === 2) {
        obj.props.rowSpan = 0;
      }

      return obj;
    },
  },
  {
    title: 'Email',
    dataIndex: 'email',
    colSpan: 0,
  },
];
const data = [
```
